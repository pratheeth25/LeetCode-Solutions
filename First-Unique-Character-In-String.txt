class Solution {
    public int firstUniqChar(String s) {
        int ans = -1;
        HashMap<Character,Integer> map1 = new HashMap<>();
        for(char c:s.toCharArray()){
            map1.put(c,map1.getOrDefault(c,0)+1);
        }
        for(int i=0;i<s.length();i++){
            char c = s.charAt(i);
            if(map1.get(c) == 1){
                ans = i;
                break;
            }
        }
        return ans;
    }
}

Calculate frequency 
check freq == 1
if present return index else -1
